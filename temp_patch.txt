          .channel("messages")
          .on(
            "postgres_changes",
            { event: "INSERT", schema: "public", table: "messages" },
            (payload: any) => {
              setMessages((prev) => [payload.new as Message, ...prev]);

          )
          .subscribe()

      if (channel) supabase.removeChannel(channel);
